CREATE TABLE PURCHASE_MASTER
(
    ID            NUMBER 	    	PRIMARY KEY,
    PURCHASE_DATE DATE			     NOT NULL,
    SUPPLIER_ID   NUMBER 		     NOT NULL,
    INVOICE_NO    VARCHAR2(30)	     NOT NULL UNIQUE,
    INVOICE_DATE  DATE			     NOT NULL,
    NOTE          VARCHAR2(4000),
    CREATE_BY     NUMBER 		    NOT NULL,
    CREATE_DATE   TIMESTAMP(6)   	DEFAULT SYSDATE,
    VERIFY_BY     NUMBER 	    	NOT NULL,
    VERIFY_DATE   TIMESTAMP(6)   	DEFAULT SYSDATE
);

CREATE TABLE SUPPLIER
(
    ID          NUMBER 		    	PRIMARY KEY,
    NAME        VARCHAR2(1000)   	NOT NULL UNIQUE,
    MOBILE      NUMBER 			    NOT NULL,
    ADDRESS     VARCHAR2(1000)  	NOT NULL,
    SHORT_NAME  VARCHAR2(1000),
    CREATE_BY   NUMBER 			    NOT NULL,
    CREATE_DATE TIMESTAMP(6) 	    DEFAULT SYSDATE
);


CREATE TABLE PRODUCT_MASTER(
    ID          	NUMBER              PRIMARY KEY,
    PRODUCT_NAME	VARCHAR2(200)       NOT NULL,
    SUPPLIER_ID 	VARCHAR2(100)       NOT NULL,
    CATEGORY_ID 	VARCHAR2(100)       NOT NULL,
    UOM_ID      	VARCHAR2(100)       NOT NULL,
    STATUS      	VARCHAR2(100)       DEFAULT 'ACTIVE',
    CREATE_BY	    NUMBER              NOT NULL,
    CREATE_DATE     TIMESTAMP(6)        DEFAULT SYSDATE 
);
